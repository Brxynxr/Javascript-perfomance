📦 Variables

Tres formas de declarar variables. La regla general: usa const por defecto, let cuando necesitas reasignar.

demo interactivo
Haz clic para ver el resultado...
// ═══ DECLARACIÓN ═══
const nombre = 'Breyner';   // no se puede reasignar
let   edad   = 20;           // sí se puede reasignar
var   ciudad = 'Barranquilla'; // evitar (scope de función, hoisting)

// const NO significa inmutable para objetos/arrays
const arr = [1, 2, 3];
arr.push(4);  // OK: el array puede mutar
// arr = [5, 6]; // ❌ ERROR: no puedes reasignar la referencia

// SCOPE (alcance)
let global = 'disponible en todo el archivo';
function ejemplo() {
  let local = 'solo dentro de la función';
  if (true) {
    let bloque = 'solo dentro del if';
  }
}

// HOISTING: var se "sube", let/const no
console.log(x);  // undefined (var hoisted)
var x = 5;
// console.log(y); // ❌ ReferenceError (let no hoisted)
let y = 5;

todos los operadores
// ARITMÉTICOS
10 + 3   // 13  suma

10 / 3   // 3.33 división

// COMPARACIÓN
5 == '5'  // true  (compara SOLO valor, convierte tipos)

!true            // false  (NOT: invierte)

// NULLISH / OPTIONAL CHAINING (ES2020)
const nombre = usuario?.nombre ?? 'Anónimo';

// ?. accede sin error si es null/undefined

// ASIGNACIÓN
x += 5;   // x = x + 5
x -= 2;   // x = x - 2
x *= 3;   // x = x * 3
x /= 4;   // x = x / 4
x ??= 'default';  // asigna solo si x es null/undefined

// TERNARIO
const msg = edad >= 18 ? 'mayor' : 'menor';

🔄 Bucles

demo — tabla de multiplicar
Escribe un número...
// FOR (cuántas veces sabes)
for (let i = 0; i < 10; i++) {
  console.log(i);
}

// WHILE (mientras condición sea true)
let i = 0;
while (i < 5) {
  console.log(i);
  i++;
}

// DO WHILE (ejecuta al menos una vez)
do {
  console.log(i);
  i++;
} while (i < 5);

// FOR...OF (iterar sobre arrays/strings)
const frutas = ['mango', 'papaya', 'maracuyá'];
for (const fruta of frutas) {
  console.log(fruta);
}

// FOR...IN (iterar sobre propiedades de objeto)
const persona = { nombre: 'Breyner', ciudad: 'Baq' };
for (const clave in persona) {
  console.log(clave + ': ' + persona[clave]);
}

// BREAK y CONTINUE
for (let i = 0; i < 10; i++) {
  if (i === 3) continue;  // salta i=3
  if (i === 7) break;     // termina en i=7
}

🎯 Funciones

demo — calculadora
Haz clic...
// DECLARACIÓN (hoisting: se puede llamar antes de declarar)
function saludar(nombre = 'mundo') {
  return 'Hola, ' + nombre;
}

// EXPRESIÓN (no tiene hoisting)
const saludar = function(nombre) {
  return `Hola, ${nombre}`;
};

// ARROW FUNCTION (ES6) ← más usada
const sumar = (a, b) => a + b;           // retorno implícito
const elevar = x => x ** 2;              // 1 param, sin paréntesis
const saludar = (nombre) => {             // con bloque
  const msg = `Hola ${nombre}!`;
  return msg;
};

// PARÁMETROS ESPECIALES
function greet(nombre = 'mundo') {}  // default
function suma(...nums) {               // rest: agrupa el resto
  return nums.reduce((a, b) => a + b, 0);
}

// CLOSURES (función que recuerda su entorno)
function contador() {
  let count = 0;
  return () => ++count;  // recuerda 'count'
}
const incrementar = contador();
incrementar();  // 1
incrementar();  // 2

// IIFE (función que se llama a sí misma)
(function() {
  console.log('se ejecuta inmediatamente');
})();

📊 Arrays y sus métodos

demo — transforma el array

Array original:

Elige una operación...
const arr = [1, 2, 3, 4, 5];

// ACCESO Y MODIFICACIÓN
arr[0]                  // 1 (primer elemento)
arr.length              // 5
arr.push(6)              // agrega al final → [1,2,3,4,5,6]
arr.pop()               // elimina y retorna el último → 6
arr.unshift(0)          // agrega al inicio → [0,1,2,3,4,5]
arr.shift()             // elimina y retorna el primero → 0
arr.splice(1, 2)        // elimina 2 desde índice 1
arr.splice(1, 0, 'a')  // inserta 'a' en índice 1

// BÚSQUEDA
arr.indexOf(3)          // 2 (posición)
arr.includes(3)        // true
arr.find(n => n > 3)   // primer elemento que cumple
arr.findIndex(n => n > 3) // índice del primer resultado

// ITERACIÓN (los más importantes)
arr.forEach(n => console.log(n))    // ejecuta función por cada elemento
arr.map(n => n * 2)                  // crea nuevo array transformado
arr.filter(n => n % 2 === 0)        // filtra elementos
arr.reduce((acc, n) => acc + n, 0)  // reduce a un valor
arr.some(n => n > 4)               // true si alguno cumple
arr.every(n => n > 0)              // true si todos cumplen
arr.sort((a, b) => a - b)           // ordena (modifica original)

// CREACIÓN DE COPIAS (sin mutar)
arr.slice(1, 3)          // copia de índice 1 a 2
[...arr]                  // copia completa con spread
arr.concat([6, 7])       // une arrays
arr.flat()               // aplana arrays anidados

// DESESTRUCTURACIÓN
const [primero, segundo, ...resto] = arr;
// primero=1, segundo=2, resto=[3,4,5]

🏗️ Objetos

objetos en profundidad
// CREACIÓN
const persona = {
  nombre: 'Breyner',
  edad: 20,
  ciudad: 'Barranquilla',
  hobbies: ['música', 'anime', 'código'],
  saludar() {            // método shorthand
    return `Hola, soy ${this.nombre}`;
  }
};

// ACCESO
persona.nombre             // 'Breyner' (dot notation)
persona['nombre']         // 'Breyner' (bracket notation)
const clave = 'edad'; persona[clave]  // dinámico

// MODIFICAR
persona.email = 'b@mail.com';  // agregar propiedad
persona.edad = 21;             // modificar
delete persona.ciudad;         // eliminar

// DESESTRUCTURACIÓN
const { nombre, edad, ciudad = 'desconocida' } = persona;
const { nombre: miNombre } = persona; // renombrar

// SPREAD (clonar / mezclar)
const copia = { ...persona };
const extendido = { ...persona, email: 'nuevo@mail.com' };

// MÉTODOS DEL OBJETO Object
Object.keys(persona)      // ['nombre', 'edad', ...]
Object.values(persona)    // ['Breyner', 20, ...]
Object.entries(persona)  // [['nombre','Breyner'], ...]
Object.assign({}, persona, { edad: 21 }) // mezcla
Object.freeze(persona)   // hace inmutable

// VERIFICAR PROPIEDAD
'nombre' in persona        // true
persona.hasOwnProperty('nombre') // true

⏳ Async / Promesas

JavaScript es single-thread. Las promesas y async/await manejan operaciones asíncronas (peticiones, timers, archivos).

demo — setTimeout y Promesa
Haz clic...
// PROMISES
const promesa = new Promise((resolve, reject) => {
  const exito = true;
  if (exito) resolve('¡Funcionó!');
  else       reject(new Error('Algo falló'));
});

promesa
  .then(resultado => console.log(resultado))
  .catch(error    => console.error(error))
  .finally(()    => console.log('siempre ejecuta'));

// ASYNC / AWAIT (syntactic sugar sobre promesas)
async function obtenerDatos() {
  try {
    const resultado = await promesaQueTardaUnRato();
    console.log(resultado);
  } catch (error) {
    console.error(error);
  }
}

// PROMESAS EN PARALELO
const [r1, r2] = await Promise.all([promesa1, promesa2]);
const primero  = await Promise.race([p1, p2]);
const resultados = await Promise.allSettled([p1, p2]);

// TIMERS
setTimeout(() => {}, 2000);          // ejecuta una vez tras 2s
const id = setInterval(() => {}, 1000); // repite cada 1s
clearInterval(id);                   // detiene el intervalo

🌐 Fetch API

demo — petición real a API pública
Haz clic para hacer la petición...
// GET (obtener datos)
const res = await fetch('https://api.ejemplo.com/datos');
const data = await res.json();

// POST (enviar datos)
const res = await fetch('/api/usuarios', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ nombre: 'Breyner', edad: 20 })
});

// MANEJO DE ERRORES
async function getData(url) {
  try {
    const res = await fetch(url);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (err) {
    console.error('Error:', err);
  }
}

// RESPONSE METHODS
res.json()   // parsea JSON
res.text()   // texto plano
res.blob()   // datos binarios (imágenes)
res.ok       // true si status 200-299
res.status   // código HTTP (200, 404, 500...)

💾 Web Storage

demo localStorage
localStorage vacío...

localStorage.setItem('usuario', 'Breyner');
localStorage.getItem('usuario');           // 'Breyner'
localStorage.removeItem('usuario');
localStorage.clear();                      // borra todo
localStorage.length;                       // cantidad de keys

const obj = { nombre: 'Breyner', nivel: 5 };
localStorage.setItem('perfil', JSON.stringify(obj));

// sessionStorage — se borra al cerrar la pestaña
sessionStorage.setItem('token', 'abc123');
sessionStorage.getItem('token');
sessionStorage.clear();

// Diferencias
// localStorage:   persiste indefinidamente
// sessionStorage: solo dura la sesión del tab
// cookies:        para enviar al servidor, tienen expiración

lo más importante de ES6 en adelante
// TEMPLATE LITERALS
const msg = `Hola ${nombre}, tienes ${edad} años`;
const html = `
  <div class="card">

const { a, b, ...resto } = objeto;       // objetos

  constructor(nombre) { this.nombre = nombre; }
